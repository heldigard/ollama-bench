```python
from __future__ import annotations

import asyncio
import random
from typing import Any, Callable, Coroutine, Optional

async def retry_with_backoff(
    fn: Callable[[], Coroutine[Any, Any, Any]],
    *,
    retries: int = 3,
    base_delay: float = 0.5,
    jitter: bool = True,
) -> Any:
    """Call an async function with exponential backoff on exception."""
    attempt = 0
    last_exception: Optional[Exception] = None

    while attempt <= retries:
        try:
            return await fn()
        except Exception as exc:
            last_exception = exc
            if attempt == retries:
                raise
            delay = base_delay * (2 ** attempt)
            if jitter and random.random() < 0.5:
                delay += random.uniform(0, delay)
            await asyncio.sleep(delay)
            attempt += 1

    # Should never reach here due to the `raise` above, but kept for type safety.
    raise RuntimeError("Unexpected exit from retry_with_backoff")
```
