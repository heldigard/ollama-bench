```python
from __future__ import annotations

import asyncio
import random
from typing import Any, Callable, TypeVar

T = TypeVar("T")

async def retry_with_backoff(
    fn: Callable[[], T], *, retries: int = 3, base_delay: float = 0.5, jitter: bool = True
) -> T:
    """Call an async function with exponential backoff on failure."""
    last_exception: Exception | None = None

    for attempt in range(retries + 1):
        try:
            return await fn()
        except Exception as exc:
            last_exception = exc
            if attempt == retries:
                raise
            delay = base_delay * (2 ** attempt)
            if jitter and random.random() < 0.5:
                delay += random.uniform(0, delay / 2)
            await asyncio.sleep(delay)

    # Fallback unreachable due to `raise` above, but kept for type safety
    raise RuntimeError("Unexpected exit from retry loop") from last_exception
```
