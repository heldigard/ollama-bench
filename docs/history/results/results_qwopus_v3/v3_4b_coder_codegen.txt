```python
from __future__ import annotations

import asyncio
import random
from typing import Awaitable, ParamSpec, TypeVar

P = ParamSpec("P")
R = TypeVar("R", covariant=True)

async def retry_with_backoff(fn: Awaitable[R], *, retries: int = 3, base_delay: float = 0.5, jitter: bool = True) -> R:
    """Call an async function with exponential backoff on Exception."""
    last_exception: Exception | None = None
    
    for attempt in range(retries + 1):
        try:
            return await fn()
        except Exception as exc:
            last_exception = exc
            if attempt == retries:
                raise
            
            delay = base_delay * (2 ** attempt)
            
            if jitter and random.random() < 0.5:
                delay *= random.uniform(0.5, 1.5)
            
            await asyncio.sleep(delay)
    
    # Should never reach here due to the `raise` above, but kept for type safety
    raise last_exception from None
```
